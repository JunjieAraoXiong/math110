\exercise
(Axler 5.B.10) Let $p$ be a polynomial and $T \colon V \to V$ be a linear map. Given an eigenvector $v \in V$ with eigenvalue $\lambda$, show that $p(T)v = p(\lambda)v$.
\ruledspace{6cm}

\exercise
Let $V$ denote the $\R$-vector space spanned by the functions $\{1, \sin x, \cos x, \sin 2x, \cos 2x\}$, and let $d \colon V \to V$ denote the differentiation operator. What are the eigenvalues of $d$? Is $d$ diagonalizable?
\ruledspace{7cm}
\answerline

%==============================================================================
% UPPER-TRIANGULAR MATRICES (3)
%==============================================================================

\exercise
\begin{enumerate}[label=(\alph*)]
    \item Define the linear map $T \colon \C^2 \to \C^2$ by $T(x, y) := (y, -x)$. What are the eigenvalues of $T$? Find a basis of $\C^2$ such that the matrix associated to $T$ is upper-triangular.
    \ruledspace{6cm}
    \answerline

    \item Define the linear map $T \colon \R^2 \to \R^2$ by $T(x, y) := (y, -x)$. Show that there is no basis of $\R^2$ such that the matrix associated to $T$ is upper-triangular.
    \ruledspace{6cm}
\end{enumerate}

%------------------------------------------------------------------------------

\exercise
(Axler 5.B.18) Fix an invertible linear map $T \colon V \to V$.

\begin{enumerate}[label=(\alph*)]
    \item If $p(x)$ is the minimal polynomial of $T$, show that $x^{\deg p} p(1/x)$ is the minimal polynomial of $T^{-1}$.
    \ruledspace{6cm}

    \item For each nonzero scalar $\lambda$, show that $E(\lambda, T) = E(\lambda^{-1}, T^{-1})$.
    \ruledspace{6cm}
\end{enumerate}

%==============================================================================
% EIGENSPACES AND RANGE (5)
%==============================================================================

\exercise
Fix a finite-dimensional vector space $V$. Fix a linear map $T \colon V \to V$ with nonzero eigenvalues $\lambda_1, \ldots, \lambda_m$.

\begin{enumerate}[label=(\alph*)]
    \item Show that $E(\lambda_i, T) \subseteq \operatorname{range} T$ for each eigenvalue $\lambda_i$.
    \ruledspace{6cm}

    \item Use (a) to conclude that
    \[
        \dim E(\lambda_1, T) + \cdots + \dim E(\lambda_m, T) \leq \dim \operatorname{range} T.
    \]
    \ruledspace{5cm}
\end{enumerate}

%==============================================================================
% GENERALIZED EIGENSPACES (6)

\exercise
Find the bases for the generalized eigenspace(s) of the following matrix:
\[
    \begin{bmatrix}
    4 & 1 & 0 \\
    1 & 4 & 1 \\
    4 & -4 & 7
    \end{bmatrix}
\]
\ruledspace{8cm}
\answerline

\exercise
