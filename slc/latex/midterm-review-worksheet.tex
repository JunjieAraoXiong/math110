%==============================================================================
% LINEAR MAPS AND DUAL SPACES (1)
%==============================================================================

\exercise
Given a finite-dimensional vector space $V$ with basis $v_1, \ldots, v_n$.

\begin{enumerate}[label=(\alph*)]
    \item Prove that if two linear maps $S$ and $T$ have $S(v_1) = T(v_1), \ldots, S(v_n) = T(v_n)$, then $S = T$.
    \ruledspace{6cm}

    \item Prove that for a linear map $T$, we have $T(v) = \varphi_1(v)T(v_1) + \cdots + \varphi_n(v)T(v_n)$ with $\varphi_1, \ldots, \varphi_n$ the dual basis of $v_1, \ldots, v_n$.
    \ruledspace{7cm}
\end{enumerate}

%==============================================================================
% DIFFERENTIATION OPERATOR (2)
%==============================================================================

\exercise
Let $V$ denote the $\R$-vector space spanned by the functions $\{1, \sin x, \cos x, \sin 2x, \cos 2x\}$, and let $d \colon V \to V$ denote the differentiation operator. What are the eigenvalues of $d$? Is $d$ diagonalizable?
\ruledspace{8cm}
\answerline

%==============================================================================
% UPPER-TRIANGULAR MATRICES (3)
%==============================================================================

\exercise
\begin{enumerate}[label=(\alph*)]
    \item Define the linear map $T \colon \C^2 \to \C^2$ by $T(x, y) := (y, -x)$. What are the eigenvalues of $T$? Find a basis of $\C^2$ such that the matrix associated to $T$ is upper-triangular.
    \ruledspace{6cm}
    \answerline

    \item Define the linear map $T \colon \R^2 \to \R^2$ by $T(x, y) := (y, -x)$. Show that there is no basis of $\R^2$ such that the matrix associated to $T$ is upper-triangular.
    \ruledspace{6cm}
\end{enumerate}
