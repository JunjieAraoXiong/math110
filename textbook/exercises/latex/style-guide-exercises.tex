%------------------------------------------------------------------------------
% WRITING PHILOSOPHY
%------------------------------------------------------------------------------
%
% PRIORITY: Clarity over brevity. These are learning exercises, not exams.
%
% 1. BE DETAILED IN SOLUTIONS:
%    - Show every step in derivations (don't skip algebra)
%    - State what you're doing before you do it ("By rank-nullity:", "Row reducing:")
%    - Use LHS/RHS labels when comparing expressions
%    - Explain WHY each step works, not just WHAT it is
%
% 2. BE CLEAR IN PROBLEM STATEMENTS:
%    - Use plain English, avoid jargon abbreviations
%    - State goals explicitly ("Show that...", "Find...", "Prove...")
%    - Include hints when appropriate
%    - Use concrete examples with actual matrices
%
% 3. STRUCTURE FOR UNDERSTANDING:
%    - Label steps: "Step 1:", "Step 2:", etc.
%    - Use parallel structure across similar problems
%    - Put key results in boxes, but show the work to get there
%    - End solutions with boxed final answers
%
% 4. WHAT TO AVOID:
%    - Cryptic abbreviations without definition
%    - Skipping "obvious" steps (nothing is obvious when learning)
%    - Dense walls of math without prose guidance
%    - Assuming the reader remembers earlier definitions
%    - Em dashes (---). Use colons, semicolons, or break into sentences instead
%

% DIRECT PROOF:
%
%   \begin{solution}
%   Let $\vect{v} \in \ker(T)$. Then $T(\vect{v}) = \vect{0}$.
%
%   Since $T$ is linear, $T(c\vect{v}) = cT(\vect{v}) = c \cdot \vect{0} = \vect{0}$.
%
%   Therefore $c\vect{v} \in \ker(T)$, so $\ker(T)$ is closed under scalar multiplication.
%
%   Similarly, for $\vect{u}, \vect{v} \in \ker(T)$:
%   $T(\vect{u} + \vect{v}) = T(\vect{u}) + T(\vect{v}) = \vect{0} + \vect{0} = \vect{0}$.
%
%   Thus $\ker(T)$ is a subspace. \hfill $\square$
%   \end{solution}
%
% PROOF BY CONTRADICTION:
%
%   \begin{solution}
%   Suppose, for contradiction, that $\vect{v}_1, \ldots, \vect{v}_n$ are
%   linearly dependent.
%
%   Then there exist $c_1, \ldots, c_n$, not all zero, such that
%   $c_1\vect{v}_1 + \cdots + c_n\vect{v}_n = \vect{0}$.
%
%   ...
%
%   This contradicts our assumption. \hfill $\lightning$
%   \end{solution}

%
%   \begin{solution}
%   \textbf{Base case:} When $n = 1$, ...

%
%   \textbf{Inductive step:} We show the result holds for $n = k + 1$.

\section*{Part I: Computation}
%==============================================================================

\begin{problem}
Let $\mat{A} = \begin{pmatrix} 2 & 1 \\ 1 & 2 \end{pmatrix}$.
\begin{enumerate}[label=(\alph*)]
    \item Find the eigenvalues of $\mat{A}$.
    \item Find a basis for each eigenspace.
    \item Is $\mat{A}$ diagonalizable? If so, find matrices $\mat{P}$ and $\mat{D}$.
\end{enumerate}
\end{problem}

\begin{solution}
\textbf{(a)} The characteristic polynomial is:
\[
    \det(\mat{A} - \lambda \mat{I}) = (2-\lambda)^2 - 1 = \lambda^2 - 4\lambda + 3 = (\lambda - 1)(\lambda - 3).
\]
So $\boxed{\lambda_1 = 1, \lambda_2 = 3}$.

\textbf{(b)} For $\lambda = 1$: Solve $(\mat{A} - \mat{I})\vect{x} = \vect{0}$.
\[
    E_1 = \spn\left\{ \begin{pmatrix} 1 \\ -1 \end{pmatrix} \right\}.
\]

For $\lambda = 3$: Solve $(\mat{A} - 3\mat{I})\vect{x} = \vect{0}$.
\[
    E_3 = \spn\left\{ \begin{pmatrix} 1 \\ 1 \end{pmatrix} \right\}.
\]

\textbf{(c)} Yes, $\mat{A}$ is diagonalizable since we have 2 linearly independent eigenvectors.
\[
    \boxed{\mat{P} = \begin{pmatrix} 1 & 1 \\ -1 & 1 \end{pmatrix}, \quad
    \mat{D} = \begin{pmatrix} 1 & 0 \\ 0 & 3 \end{pmatrix}}
\]
\end{solution}
